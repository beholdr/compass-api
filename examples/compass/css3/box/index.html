scss sass css 
@import "compass/css3";

.example {
  @include display-box;
  @include box-orient(horizontal);
  @include box-align(stretch);
  @include box-direction(reverse);
  height: 200px;
  width: 100%;
  .content, .sidebar {
    padding: 20px;
  }
  .content {
    @include box-flex(4);
    background-color: #c5c1b4;
  }
  .sidebar {
    @include box-flex(1);
    background-color: #375f99;
    margin-right: 1px;
  }
}
