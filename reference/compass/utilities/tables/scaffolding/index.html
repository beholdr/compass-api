table-scaffolding 

=table-scaffolding
  th

  td,
  th
    padding: 2px
    &.numeric
      text-align: right

@mixin table-scaffolding {
  th {
    text-align: center;
    font-weight: bold;
  }
  td,
  th {
    padding: 2px;
    &.numeric {
      text-align: right;
    }
  }
}
